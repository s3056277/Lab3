        // Example JavaScript for addTodo function
        let todoList = [];
 
        function addTodo() {
            const todoInput = document.getElementById('todoInput');
            if (todoInput == null) {
                console.error('The todo input is missing from the page!');
                return;
            }
 
            const newTodo = todoInput.value;
 
            // verify that there is text
            if ('' !== newTodo.trim()) {
                console.log('Adding todo: ', newTodo);
 
                // add the new item to the list
                todoList.push(newTodo);
 
                // keep the list sorted
                todoList.sort();
 
                console.log('New todo list: ', todoList);
 
                // clear the input
                todoInput.value = '';
 
                // update the todo list display
                updateTodoList();
 
                // apply the todo list filter
                filterTodoList();
            }
        }
 
        function updateTodoList() {
            const container = document.getElementById('todoListContainer');
            if (todoList.length === 0) {
                container.innerHTML = 'Nothing to do, hurray!';
            } else {
                container.innerHTML = '<ul>' + todoList.map(item => `<li>${item}</li>`).join('') + '</ul>';
            }
        }
 
        function removeTodoListItem(itemToRemove) {
            console.log("item to remove: ", itemToRemove);
 
            todoList = todoList.filter((value) => {
                return value !== itemToRemove;
            });
            // unsafe alternative: todoList.splice(...)
 
            // update the todo list
            updateTodoList();
 
            // apply the todo filter
            filterTodoList();
        }
